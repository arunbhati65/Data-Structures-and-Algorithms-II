char[] reverse(char[] arr){
	int l=0,h=arr.length-1;
	while(l<=h){
		char temp=arr[l];
		arr[l]=arr[h];
		arr[h]=temp;
		++l;
		--h;
	}
	return arr;
}

int maximumSum(int[] arr){
	int[] memo=new int[arr.length];
	memo[0]=arr[0];
	int maxSum=memo[0];
	for(int i=1;i<arr.length;++i){
		memo[i]=Math.max(arr[i],arr[i]+memo[i-1]);
		maxSum=Math.max(maxSum.memo[i]);
	}
	return maxSum;
}
